数据库操作命令速查.

增

insert into man (id, name, age) value (1, 'jack', 27)
# 按照结构插入数据

insert man values (4, 'a', 1), (5, 'b', null)
# 插入多条数据

删

delete form man
delete * form man
# 删除整张表

delete from man where id = 2
# 删除表中的指定数据

truncate man
#不可逆删除

改

update man set id = 2 where name = 'jack'
# 更新 man 表中为 jack 的数据，将 id 改为 2

update man set age = 30 where id = 5 or id = 4
# 更新 man 表中 id 为 4 或 5 的数据，将 age 改为 30

查

desc man
# 看 man 表的数据格式

desc bus.man
# 看 bus 库的 man 表的数据格式

select * from man
# 查询 man 表全部的数据

select age from man
# 查询 man 表中的 age 列数据

select name as na, age as ag from man
# 查数据的时候设置别名，name 设为 na，age 设为 ag

select * from man where age < 50
select * from man where age != 50
select * from man where age is null
select * from man where age is not null
select * from man where age between 27 and 31
select * from man where name like '%a%'
# 查询符合条件的数据，%a% 表示含有 a 的字符串

select * from man order by age
select * from man order by age limit 3
# 升序查询，limit 可限制显示数量

select * from man order by age desc
select * from man order by age desc limit 3
# 降序查询，limit 可限制显示数量

其他

show character set
# 显示支持的字符集

show tables from bus
# 显示 bus 包含的所有表

show grants
# 查看授权

show grants for root
# 查看指定用户的授权

grant all on *.* to 'hzy'@'192.168.207.131' identified by 'hzy123'
# 授权用户在数据库本机上登录访问所有数据库

revoke all on "." from 'root'@'%'
# 移除授权
